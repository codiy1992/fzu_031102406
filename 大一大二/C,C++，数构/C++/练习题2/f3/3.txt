题目描述：设计一个学生的类，成绩是其私有数据成员。 读入一组学生的成绩，设计一个计算这些学生平均成绩的友元函数，输出学生的平均成绩。要求采用C++面向对象的方法实现。 
输入格式： 测试输入包含若干测试用例，每个测试用例占一行，每一行的第一个数表示学生个数（学生个数小于20），后面的数字表示每个学生的成绩。每个数字间用空格隔开。当读入0时(即学生个数为0)输入结束，相应的结果不要输出。
输出格式：对每个测试用例输出一行。每个输出结果的精度，保留小数点后面4位。
输入样例：
3 89 97 85.5
5 60 50 80 90 100
0
输出样例：
90.5000
76.0000
退出程序结束






#include <iostream>
#include <iomanip>
using namespace std;
class student
{
private:
	float number;
public:
	friend void display();
}stu1;

void display()
{
	int n,i,j;
	float sum;
	for(j=0;;j++)
	{
		cin>>n;
		if(n==0) break;
		sum=0;
		for(i=0;i<n;i++)
		{
			cin>>stu1.number;
			sum=sum+stu1.number;
		}
		float average;
		average=sum/n;
		cout<<setiosflags(ios::fixed)<<setprecision(4)<<average<<endl;
	}
}
int main()
{
	display();
}
